# Parte 7: Función Principal
def calculadora():
    while True:
        mostrar_menu()
        operacion = obtener_operacion()

        if operacion in ('1', '3'):
            num1, num2 = obtener_numeros()
            resultado = realizar_operacion(operacion, num1, num2)
            print("El resultado es:", resultado)
        elif operacion in ('2', '4'):
            num1, num2 = obtener_numeros_resta_division()
            resultado = realizar_operacion(operacion, num1, num2)
            print("El resultado es:", resultado)
        elif operacion == '5':
            num = float(input("Ingrese el número para calcular la raíz cuadrada: "))
            resultado = realizar_raiz_cuadrada(num)
            print("El resultado es:", resultado)
        elif operacion == '6':
            base = float(input("Ingrese la base: "))
            exponente = float(input("Ingrese el exponente: "))
            resultado = realizar_elevacion_potencia(base, exponente)
            print("El resultado es:", resultado)
        elif operacion == '7':
            print("¡Hasta pronto, gracias por usar la calculadora SB!")
            break
        else:
            print("Operación no válida. Por favor, seleccione un número del 1 al 7.")

if __name__ == "__main__":
    calculadora()
